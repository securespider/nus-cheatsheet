\subsection{Newsvendor Problem}
\subsubsection{Problem statement}
\begin{itemize}
	\item Newsvendor sells newspaper everyday
	\item Vendor buy at \$.5 of $y$ quantity
	\item Vendor sells min($D_{demand}$, $y$) newspaper @ \$2 
\end{itemize}
\subsubsection{Optimal solution}
\begin{description}
	\item[\hl{Overage}]{Buying more papers than demand $c_o$}
	\item[Underage]{Buying less paper than demand $c_u$}
\end{description}
\begin{itemize}
	\item Buying more based on phi inverse
	\item $y^{\star}=\mu + \sigma \phi^{-1}(\dfrac{c_u}{c_u+c_o})$
	\item Out of sample validation
\end{itemize}
\section{Optimization}
\begin{itemize}
	\item Select a decision that max/min objective
	\item $Z^*=min(f(x)) ~~|~~ x\in\chi$
	\item $max(f(x)) = - min(-f(x))$
	\item Optimization problem can be
	\begin{description}
		\item[Infeasible]{Sample set is $\null$}
		\item[Unbounded]{$Z^*$ is continuously decreasing}
		\item[Exists]{$f(x)\geq M$ and $x^*\in$sample where $f(x^*)=Z^*$}
	\end{description}
\end{itemize}

\section{04. Convex Optimization}
\begin{itemize}
	\item \keyword{Convex set}{$\forall x,y\in S, \lambda\in[0,1]$, $\lambda x + (1-\lambda)y\in S$}
	\item \keyword{Convex function}{$X$ is convex set and $\forall x,y\in X, \lambda\in[0,1], f(\lambda x+(1-\lambda)y)\leq \lambda f(x)+(1-\lambda)f(y)$}
	\item \keyword{epigraph}{$epi(f) = {(x,y)\in X: y\geq f(x)}$ Everything above the graph}
	\begin{itemize}
		\item Function f is convex iff epigraph is convex
		\item If f is one dimensional and twice differentiable, f(x) is convex iff $f’’(x)\geq 0$
		\item For convex functions f,g $rightarrow$ f+g, max{f,g} are convex
		\item All local minimum in convex functions are global minimum
	\end{itemize}
\end{itemize}

\section{05. Simple Linear Regression}
\subsection{Least Square Solution}
\begin{itemize}
	\item Using the sum of squares for $y_i-\beta_0-\beta_1*x_i$
	\item Minimization problem for sum of square where derivative = 0
	\item Measures unbiased estimate (converges to actual distribution)
\end{itemize}
\subsection{Hypothesis test}
\begin{itemize}
	\item Using P value to derive confidence of an association (statistical significance)
	\item Statistic follows t-distribution with n-m-1 degrees of freedom
	\begin{itemize}
		\item n = number of observations m = num features
	\end{itemize}
	\item $t=\dfrac{\beta_1}{SE(\beta_1)}$
\end{itemize}
\subsection{Qualitative predictor}
\begin{itemize}
	\item X is discrete instead of continous 
	\item Can be split up into more x1,x2.. to take 1/0 values which affects result
	\item x can take on more than 2 values (0,1,2..) to take on different observations
	\begin{itemize}
		\item Useful when observation has some topological ordering
	\end{itemize}
\end{itemize}
\subsection{Linear Regression Assumptions}
\begin{description}
	\item[Additivity]{Changes of each predictor is independent of other predictor values}
	\item[Linearity]{Unit change in X has constant change in Y (regardless of X)}
\end{description}
\subsubsection{Variable dependence}
\begin{itemize}
	\item IRL, variables affect each other
	\item Model the interaction by putting \keyword{interaction term}{product of variables}
	\item Variables are no longer linear but model is linear for beta
	\item Coefficient of interaction term (once normalised) can tell the extent of interaction
\end{itemize}
\begin{description}
	\item[Hierachical principle]{If include interaction term, the main effects/variables must be included}
\end{description}
\subsubsection{Non linear relationship}
\begin{description}
	\item[Taylor expansion]{Any function within some domain can be modelled as a high order polynomial}
\end{description}
\begin{itemize}
	\item P-values cannot be used where variables are highly correlated with each other
\end{itemize}
