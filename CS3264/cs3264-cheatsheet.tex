\section{01. Intro}
\subsection{What is}
\begin{enumerate}
	\item Supervised Learning - Classification/Regression
	\item Unsupervised - Kmeans, PCA
	\item Reinforcement - Giving rewards and punishment
\end{enumerate}
\subsection{Linear regression}
\begin{itemize}
	\item Data - tuples of points $\{(x_n, t_n)\}_{0\rightarrow N}$
	\item Model - $y=w_0+xw_1$
	\item Loss - Mean square error $\dfrac{1}{2}\sum_{n=1}^{N}(y_n-t_n)^2$
	\item Derivative to find lowest point to find $w$
	\item $t_n = [x_n][w_n]^T$
\end{itemize}
\subsection{Non-Linear regression}
\begin{itemize}
	\item Basis factor expansion $x \rightarrow [x, x^2, x^3..,x^P]$
	\item Find weights for each polynomial/exponential/..
\end{itemize}
\subsection{Regularization}
\subsubsection{Ridge regression solution}
\begin{itemize}
	\item Add regularizer ($\dfrac{\lambda}{2}||w||^2$) to error function
	\begin{itemize}
		\item Lambda is a constant 
		\item smaller = error matters more		
		\item larger = penalise weights more
	\end{itemize}
\end{itemize}
